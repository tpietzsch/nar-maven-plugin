14  /**
15   * Adds the ability to run arbitrary command line tools to post-process the
16   * compiled output (ie: ranlib/ar/etc)
17   *
18   * @author Richard Kerr
19   * @goal nar-process-libraries
20   * @phase process-classes
21   * @requiresSession
22   * @requiresProject
23   * @author Richard Kerr
24   */
25  public class NarProcessLibraries extends AbstractCompileMojo {
26  
27      /**
28       * List of commands to execute
29       *
30       * @parameter expression=""
31       */
32      private List<ProcessLibraryCommand> commands;
33  
34      private Log log = getLog();
35  
36      @Override
37      protected List getArtifacts() {
38          //TODO: Added to get code compiling after rebasing.  Should this have a concrete implementation in AbstractCompileMojo?
39          return null;
40      }

42      @Override
43      public void narExecute() throws MojoFailureException, MojoExecutionException {
44          log.info("Running process libraries");
45          // For each of the libraries defined for this build
46          for (Library library : getLibraries()) {
47              log.info("Processing library " + library);
48              String type = library.getType();
49              File outFile;
50              // Find what the output directory is
51              if (type.equalsIgnoreCase(Library.EXECUTABLE)) {
52                  File outDir = getLayout().getBinDirectory(getTargetDirectory(), getMavenProject().getArtifactId(),
53                          getMavenProject().getVersion(), getAOL().toString());
54                  outFile = new File(outDir, getOutput(false));
55              } else {
56                  File outDir = getLayout().getLibDirectory(getTargetDirectory(), getMavenProject().getArtifactId(),
57                          getMavenProject().getVersion(), getAOL().toString(), type);
58                  outFile = new File(outDir, getOutput(true));
59              }
60  
61              // Then run the commands that are applicable for this library type
62              for (ProcessLibraryCommand command : commands == null ? new ArrayList<ProcessLibraryCommand>() : commands) {
63                  if (command.getType().equalsIgnoreCase(type))
64                      runCommand(command, outFile);
65              }
66          }
67  
68      }
69  
70      private void runCommand(ProcessLibraryCommand command, File outputFile) throws MojoFailureException,
71              MojoExecutionException {
72          ProcessBuilder p = new ProcessBuilder(command.getCommandList());
73          p.command().add(outputFile.toString());
74          p.redirectErrorStream(true);
75          log.info("Running command \"" + p.command() + "\"");
76          try {
77              Process process = p.start();
78              BufferedInputStream bis = new BufferedInputStream(process.getInputStream());
79              byte[] buffer = new byte[1024];
80              int endOfStream = 0;
81              do {
82                  endOfStream = bis.read(buffer);
83                  log.debug(new String(buffer, 0, endOfStream == -1 ? 0 : endOfStream));
84              } while (endOfStream != -1);
85  
86              if (process.waitFor() != 0) {
87                  // TODO: Maybe this shouldn't be an exception, it might have
88                  // still worked?!
89                  throw new MojoFailureException("Process exited abnormally");
90              }
91          } catch (IOException e) {
92              e.printStackTrace();
93              throw new MojoFailureException("Failed to run the command \"" + p.command() + "\"", e);
94          } catch (InterruptedException e) {
95              e.printStackTrace();
96              throw new MojoFailureException("Failed to run the command \"" + p.command() + "\"", e);
97          }
98      }
